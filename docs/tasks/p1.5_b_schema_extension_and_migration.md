```sql
-- 1. virtual_series 表（P1.5-C 回填，P1.5-D 轮询）
create table if not exists virtual_series (
    id integer primary key autoincrement,
    tmdb_tv_id text not null,
    name text not null,
    original_name text,
    tmdb_status text,                  -- Returning Series / Ended / Canceled / In Production / Pilot / Planned
    tmdb_number_of_seasons integer,    -- TMDb 当前已知的最高季编号
    local_max_season integer,          -- 本系统/A 库内已知最高季
    last_polled_at text,               -- 上次轮询时间（NULL 表示从未轮询）
    poll_priority text default 'normal', -- urgent / normal / low / skip（按 tmdb_status 派生）
    created_at text not null default current_timestamp,
    updated_at text not null default current_timestamp
);

create unique index if not exists ux_virtual_series_tmdb_tv_id
on virtual_series(tmdb_tv_id);

create index if not exists idx_virtual_series_poll_priority_last_polled
on virtual_series(poll_priority, last_polled_at);

-- 2. canonical_items 新增 virtual_series_id 列（P1.5-C 回填）
alter table canonical_items add column virtual_series_id integer
    references virtual_series(id);

create index if not exists idx_canonical_items_virtual_series_id
on canonical_items(virtual_series_id);

-- 3. content_updates 新增 match_confidence_low 列（P1.5-E 写入）
alter table content_updates add column match_confidence_low integer not null default 0;
-- SQLite 用 integer 表 boolean（0=false, 1=true）

create index if not exists idx_content_updates_match_confidence_low
on content_updates(match_confidence_low);

-- 4. （根据 Q2 决策）旧字段处理
-- 选项 a：保留旧字段，由 P1.5-E 在写入逻辑中停止写入
-- 选项 b：drop column（SQLite 需要表重建，复杂度高，建议选 a）

-- 5. 标记 schema_version
insert or ignore into schema_migrations(version) values (5);
```

**注：** Q2 选 (a) 时 SQL 如上；选 (b) 时需要在本 migration 中加入 `content_updates` 表重建（CREATE NEW → COPY → DROP OLD → RENAME）的完整序列。

## 风险点

1. **风险 1：SQLite `alter table add column` 与外键约束的兼容性。** SQLite 对外键的支持需要 `pragma foreign_keys = on`，已在 `connect_database()` 中启用。但 `alter table ... references` 在某些 SQLite 版本下不会真正建立外键约束，仅作为语法记录。**对策：** 在 v5 migration 中确认 Python 3.12 自带的 sqlite3 版本（≥ 3.31）支持外键约束的 alter table；如不支持，回退到"表重建"方案。
2. **风险 2：现有生产 DB 可能已有 v4 schema 的"实际数据偏离 schema"现象。** 例如 `canonical_items.parent_canonical_item_id` 字段在 schema 里有，但实际无人写入（grep 已确认 V1 代码不写）。这不影响本任务，但应在日报中提及"v5 migration 不依赖 parent_canonical_item_id 的实际值"。
3. **风险 3：Q1/Q2 决策不在系统侧可控。** Q1 是飞书数据如何处理，Q2 是 SQLite 旧字段如何处理。这两个问题必须在任务包开工前澄清，否则 migration SQL 写法不确定。
4. **风险 4：飞书表实际字段名 vs 文档定义的字段名可能不一致。** V1 实施时，飞书侧字段名可能用了中文（如"审核状态"对应 `review_status`），文档定义只反映"系统侧字段名"。运营在飞书 UI 操作时需对照实际表头。**对策：** P1.5-B 在飞书 schema 指导文档中列出"系统字段名 ↔ 飞书显示名"的推测对照，让运营核对。
5. **风险 5：已交付的 V1 测试中可能有"硬编码 v4 schema_version"的测试。** P1.5-B migration 升至 v5 后，可能误伤这些测试。**对策：** 全量回归 `pytest tests/` 验证；如有失败，定位到具体测试后**修正测试（让它接受 v5）**，不回退 migration。
